235 inline int GetVectorBytes(DataType dtype) {
236  int data_bits = dtype.bits() * dtype.lanes();
237  // allow bool to exist
238  if (dtype == DataType::Bool() ||
239  dtype == DataType::Int(4) ||
240  dtype == DataType::UInt(4) ||
241  dtype == DataType::Int(1)) {
242  return 1;
243  }
244  CHECK_EQ(data_bits % 8, 0U)
245  << "Need to load/store by multiple of bytes";
246  return data_bits / 8;
247 }

362 inline DLDataType String2DLDataType(std::string s) {
363  DLDataType t;
364  // handle None type
365  if (s.length() == 0) {
366  t.bits = 0; t.lanes = 0; t.code = kTVMOpaqueHandle;
367  return t;
368  }
369  t.bits = 32; t.lanes = 1;
370  const char* scan;
371  if (s.substr(0, 3) == "int") {
372  t.code = kDLInt; scan = s.c_str() + 3;
373  } else if (s.substr(0, 4) == "uint") {
374  t.code = kDLUInt; scan = s.c_str() + 4;
375  } else if (s.substr(0, 5) == "float") {
376  t.code = kDLFloat; scan = s.c_str() + 5;
377  } else if (s.substr(0, 6) == "handle") {
378  t.code = kTVMOpaqueHandle;
379  t.bits = 64; // handle uses 64 bit by default.
380  scan = s.c_str() + 6;
381  } else if (s == "bool") {
382  t.code = kDLUInt;
383  t.bits = 1;
384  t.lanes = 1;
385  return t;
386  } else if (s.substr(0, 6) == "custom") {
387  t.code = ParseCustomDatatype(s, &scan);
388  } else {
389  scan = s.c_str();
390  LOG(FATAL) << "unknown type " << s;
391  }
392  char* xdelim; // emulate sscanf("%ux%u", bits, lanes)
393  uint8_t bits = static_cast<uint8_t>(strtoul(scan, &xdelim, 10));
394  if (bits != 0) t.bits = bits;
395  char* endpt = xdelim;
396  if (*xdelim == 'x') {
397  t.lanes = static_cast<uint16_t>(strtoul(xdelim + 1, &endpt, 10));
398  }
399  CHECK(endpt == s.c_str() + s.length()) << "unknown type " << s;
400  return t;
401 }

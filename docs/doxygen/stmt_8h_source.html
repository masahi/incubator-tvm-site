636  template<typename ...Args>
637  static Stmt Flatten(Args&&... seq_args) {
638  Array<Stmt> seq;
639  runtime::detail::for_each(
640  Flattener(&seq), std::forward<Args>(seq_args)...);
641  if (seq.size() == 1) return seq[0];
642  return SeqStmt(seq);
643  }
645  class Flattener {
646  public:
647  explicit Flattener(Array<Stmt>* seq)
648  : seq_(seq) {}
649 
650  void operator()(size_t i, const Stmt& stmt) const {
651  if (!stmt.defined()) return;
652  if (auto* op = stmt.as<SeqStmtNode>()) {
653  operator()(0, op->seq);
654  } else {
655  seq_->push_back(stmt);
656  }
657  }
658 
659  template<typename T>
660  void operator()(size_t i, const T& seq) const {
661  for (auto v : seq) {
662  this->operator()(0, v);
663  }
664  }
665 
666  private:
667  Array<Stmt>* seq_;
668  };

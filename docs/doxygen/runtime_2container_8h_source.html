109 template <typename ArrayType, typename ElemType>
110 class InplaceArrayBase {
111  public:
117  const ElemType& operator[](size_t idx) const {
118  size_t size = Self()->GetSize();
119  CHECK_LT(idx, size) << "Index " << idx << " out of bounds " << size << "\n";
120  return *(reinterpret_cast<ElemType*>(AddressOf(idx)));
121  }
122 
128  ElemType& operator[](size_t idx) {
129  size_t size = Self()->GetSize();
130  CHECK_LT(idx, size) << "Index " << idx << " out of bounds " << size << "\n";
131  return *(reinterpret_cast<ElemType*>(AddressOf(idx)));
132  }
133 
137  ~InplaceArrayBase() {
138  if (!(std::is_standard_layout<ElemType>::value &&
139  std::is_trivial<ElemType>::value)) {
140  size_t size = Self()->GetSize();
141  for (size_t i = 0; i < size; ++i) {
142  ElemType* fp = reinterpret_cast<ElemType*>(AddressOf(i));
143  fp->ElemType::~ElemType();
144  }
145  }
146  }
147 
148  protected:
159  template <typename... Args>
160  void EmplaceInit(size_t idx, Args&&... args) {
161  void* field_ptr = AddressOf(idx);
162  new (field_ptr) ElemType(std::forward<Args>(args)...);
163  }
164 
165  private:
171  inline ArrayType* Self() const {
172  return static_cast<ArrayType*>(const_cast<InplaceArrayBase*>(this));
173  }
174 
181  void* AddressOf(size_t idx) const {
182  static_assert(alignof(ArrayType) % alignof(ElemType) == 0 &&
183  sizeof(ArrayType) % alignof(ElemType) == 0,
184  "The size and alignment of ArrayType should respect "
185  "ElemType's alignment.");
186 
187  size_t kDataStart = sizeof(ArrayType);
188  ArrayType* self = Self();
189  char* data_start = reinterpret_cast<char*>(self) + kDataStart;
190  return data_start + idx * sizeof(ElemType);
191  }
192 };

578 inline int String::memncmp(const char* lhs, const char* rhs, size_t lhs_count,
579  size_t rhs_count) {
580  if (lhs == rhs && lhs_count == rhs_count) return 0;
581 
582  for (size_t i = 0; i < lhs_count && i < rhs_count; ++i) {
583  if (lhs[i] < rhs[i]) return -1;
584  if (lhs[i] > rhs[i]) return 1;
585  }
586  if (lhs_count < rhs_count) {
587  return -1;
588  } else if (lhs_count > rhs_count) {
589  return 1;
590  } else {
591  return 0;
592  }
593 }

628 class Optional : public ObjectRef {
629  public:
630  using ContainerType = typename T::ContainerType;
631  static_assert(std::is_base_of<ObjectRef, T>::value,
632  "Optional is only defined for ObjectRef.");
633  // default constructors.
634  Optional() = default;
635  Optional(const Optional<T>&) = default;
636  Optional(Optional<T>&&) = default;
637  Optional<T>& operator=(const Optional<T>&) = default;
638  Optional<T>& operator=(Optional<T>&&) = default;
644  explicit Optional(ObjectPtr<Object> ptr) : ObjectRef(ptr) {}
645  // nullptr handling.
646  // disallow implicit conversion as 0 can be implicitly converted to nullptr_t
647  explicit Optional(std::nullptr_t) {}
648  Optional<T>& operator=(std::nullptr_t) {
649  data_ = nullptr;
650  return *this;
651  }
652  // normal value handling.
653  Optional(T other) // NOLINT(*)
654  : ObjectRef(std::move(other)) {
655  }
656  Optional<T>& operator=(T other) {
657  ObjectRef::operator=(std::move(other));
658  return *this;
659  }
660  // delete the int constructor
661  // since Optional<Integer>(0) is ambiguious
662  // 0 can be implicitly casted to nullptr_t
663  explicit Optional(int val) = delete;
664  Optional<T>& operator=(int val) = delete;
669  T value() const {
670  CHECK(data_ != nullptr);
671  return T(data_);
672  }
677  T value_or(T default_value) const {
678  return data_ != nullptr ? T(data_) : default_value;
679  }
681  explicit operator bool() const {
682  return *this != nullptr;
683  }
684  // operator overloadings
685  bool operator==(std::nullptr_t) const {
686  return data_ == nullptr;
687  }
688  bool operator!=(std::nullptr_t) const {
689  return data_ != nullptr;
690  }
691  auto operator==(const Optional<T>& other) const {
692  // support case where sub-class returns a symbolic ref type.
693  using RetType = decltype(value() == other.value());
694  if (same_as(other)) return RetType(true);
695  if (*this != nullptr && other != nullptr) {
696  return value() == other.value();
697  } else {
698  // one of them is nullptr.
699  return RetType(false);
700  }
701  }
702  auto operator!=(const Optional<T>& other) const {
703  // support case where sub-class returns a symbolic ref type.
704  using RetType = decltype(value() != other.value());
705  if (same_as(other)) return RetType(false);
706  if (*this != nullptr && other != nullptr) {
707  return value() != other.value();
708  } else {
709  // one of them is nullptr.
710  return RetType(true);
711  }
712  }
713  auto operator==(const T& other) const {
714  using RetType = decltype(value() == other);
715  if (same_as(other)) return RetType(true);
716  if (*this != nullptr) return value() == other;
717  return RetType(false);
718  }
719  auto operator!=(const T& other) const {
720  return !(*this == other);
721  }
722  template<typename U>
723  auto operator==(const U& other) const {
724  using RetType = decltype(value() == other);
725  if (*this == nullptr) return RetType(false);
726  return value() == other;
727  }
728  template<typename U>
729  auto operator!=(const U& other) const {
730  using RetType = decltype(value() != other);
731  if (*this == nullptr) return RetType(true);
732  return value() != other;
733  }
734  static constexpr bool _type_is_nullable = true;
735 };
